# ======= é…ç½® =======
class GPTConfig:
    def __init__(
        self,
        vocab_size: int = 50257,  # è¯è¡¨å¤§å°
        block_size: int = 128,    # æœ€å¤§åºåˆ—é•¿åº¦ T
        n_layer: int = 4,         # Transformer å±‚æ•°
        n_head: int = 8,          # æ³¨æ„åŠ›å¤´æ•°
        n_embd: int = 512,        # æ¨¡å‹ç»´åº¦ d_model
        dropout: float = 0.1,     # dropout æ¦‚ç‡
        tie_weights: bool = True, # LM å¤´ä¸è¯åµŒå…¥æƒé‡å…±äº«ï¼ˆGPT-2 é£æ ¼ï¼‰
        debug: bool = False,      # è°ƒè¯•å½¢çŠ¶æ‰“å°

        self.block_size = block_size

# ======= é®ç½©å¤šå¤´è‡ªæ³¨æ„åŠ› =======
class CausalSelfAttention(nn.Module):
    def __init__(self, config: GPTConfig):
        super().__init__()
        assert config.n_embd % config.n_head == 0, "n_embd å¿…é¡»èƒ½æ•´é™¤ n_head"
        self.n_head = config.n_head
        self.head_dim = config.n_embd // config.n_head
        self.dropout = config.dropout
        self.debug = config.debug

        # å°†è¾“å…¥ä¸€æ¬¡æ€§çº¿æ€§æ˜ å°„ä¸º Q,K,Vï¼ˆç»´åº¦ 3*n_embdï¼‰ï¼Œç„¶åå†æ‹†åˆ†
        self.qkv = nn.Linear(config.n_embd, 3 * config.n_embd, bias=False)
        # æ³¨æ„åŠ›è¾“å‡ºçš„æŠ•å½±ï¼ˆå¯¹åº”è®ºæ–‡é‡Œçš„ W_Oï¼‰
        self.proj = nn.Linear(config.n_embd, config.n_embd, bias=False)

        # é¢„ç”Ÿæˆå› æœé®ç½©ï¼Œä¸‹ä¸‰è§’ä¸º Trueï¼ˆå…è®¸çœ‹ï¼‰ï¼Œä¸Šä¸‰è§’ä¸º Falseï¼ˆå±è”½ï¼‰
        mask = torch.tril(torch.ones(config.block_size, config.block_size, dtype=torch.bool))
        # å½¢çŠ¶ [1,1,T,T] ä»¥ä¾¿å¹¿æ’­åˆ° [B,h,T,T]
        self.register_buffer("attn_mask", mask.view(1, 1, config.block_size, config.block_size), persistent=False)

        self.attn_drop = nn.Dropout(self.dropout) 
        self.resid_drop = nn.Dropout(self.dropout)

    def forward(self, x: torch.Tensor):
        # x: [B, T, d_model]
        B, T, C = x.shape
        if self.debug:
            print(f"[Attn] x: {x.shape}")

        # ä¸€æ¬¡æ€§å¾—åˆ° q,k,v: [B, T, 3*d_model]
        qkv = self.qkv(x)
        if self.debug:
            print(f"[Attn] qkv: {qkv.shape}")

        # åˆ‡åˆ†å¹¶é‡æ’ä¸ºå¤šå¤´ï¼šq/k/v: [B, n_head, T, head_dim]
        q, k, v = qkv.split(C, dim=2)
        q = q.view(B, T, self.n_head, self.head_dim).transpose(1, 2)  # [B,h,T,d]
        k = k.view(B, T, self.n_head, self.head_dim).transpose(1, 2)  # [B,h,T,d]
        v = v.view(B, T, self.n_head, self.head_dim).transpose(1, 2)  # [B,h,T,d]
        if self.debug:
            print(f"[Attn] q: {q.shape}, k: {k.shape}, v: {v.shape}")

        # æ³¨æ„åŠ›åˆ†æ•°: [B,h,T,T]
        att = (q @ k.transpose(-2, -1)) / math.sqrt(self.head_dim)

        # å› æœé®ç½©ï¼šåªå…è®¸çœ‹å½“å‰ä½ç½®åŠå·¦ä¾§
        mask = self.attn_mask[:, :, :T, :T]  # è£å‰ªåˆ°å½“å‰ T
        att = att.masked_fill(~mask, float("-inf"))

        # softmax -> dropout
        att = F.softmax(att, dim=-1)
        att = self.attn_drop(att)
        if self.debug:
            print(f"[Attn] att(sftmx): {att.shape}, sum over last dimâ‰ˆ1 -> {att[0,0,0,:5].sum().item():.3f}")

        # åŠ æƒæ±‚å’Œæ‹¿åˆ°è¾“å‡ºï¼š [B,h,T,d]
        y = att @ v

        # åˆå¹¶å¤´ï¼š [B,T,h*d]
        y = y.transpose(1, 2).contiguous().view(B, T, C)

        # è¾“å‡ºæŠ•å½± + dropout
        y = self.resid_drop(self.proj(y))
        if self.debug:
            print(f"[Attn] out: {y.shape}")
        return y


# ======= ä¸¤å±‚ MLPï¼ˆGELUï¼‰ =======
class MLP(nn.Module):
    def __init__(self, config: GPTConfig):
        super().__init__()
        self.fc1 = nn.Linear(config.n_embd, 4 * config.n_embd)  # æ‰©å¼ 
        self.fc2 = nn.Linear(4 * config.n_embd, config.n_embd)  # æŠ•å›
        self.act = nn.GELU()
        self.drop = nn.Dropout(config.dropout)
        self.debug = config.debug

    def forward(self, x):
        if self.debug:
            print(f"[MLP] in: {x.shape}")
        x = self.fc1(x)
        x = self.act(x)
        x = self.drop(x)
        x = self.fc2(x)
        x = self.drop(x)
        if self.debug:
            print(f"[MLP] out: {x.shape}")
        return x


# ======= Transformer Blockï¼ˆé¢„å½’ä¸€åŒ–ï¼‰ =======
class Block(nn.Module):
    def __init__(self, config: GPTConfig):
        super().__init__()
        self.ln1 = nn.LayerNorm(config.n_embd)
        self.attn = CausalSelfAttention(config)
        self.ln2 = nn.LayerNorm(config.n_embd)
        self.mlp = MLP(config)
        self.debug = config.debug

    def forward(self, x):
        if self.debug:
            print(f"[Block] in: {x.shape}")
        # æ³¨æ„åŠ›å­å±‚ï¼ˆPre-LNï¼‰
        x = x + self.attn(self.ln1(x))
        # MLP å­å±‚ï¼ˆPre-LNï¼‰
        x = x + self.mlp(self.ln2(x))
        if self.debug:
            print(f"[Block] out: {x.shape}")
        return x


# ======= GPT ä¸»ä½“ =======
class GPT(nn.Module):
    def __init__(self, config: GPTConfig):
        super().__init__()
        self.config = config

        # è¯åµŒå…¥ & ç»å¯¹ä½ç½®åµŒå…¥ï¼ˆGPT-2 é£æ ¼ï¼‰
        self.wte = nn.Embedding(config.vocab_size, config.n_embd)       # token embedding
        self.wpe = nn.Embedding(config.block_size, config.n_embd)       # position embedding

        self.drop = nn.Dropout(config.dropout)

        # N å±‚ Block
        self.h = nn.ModuleList([Block(config) for _ in range(config.n_layer)])

        # æœ€ç»ˆ LayerNorm
        self.ln_f = nn.LayerNorm(config.n_embd)

        # è¯è¡¨æŠ•å½±å¤´
        self.lm_head = nn.Linear(config.n_embd, config.vocab_size, bias=False)

        # æƒé‡å…±äº«ï¼šlm_head.weight ä¸ wte.weight ç»‘å®šï¼ˆGPT-2 ä¹ æƒ¯ï¼‰
        if config.tie_weights:
            self.lm_head.weight = self.wte.weight

        # å‚æ•°åˆå§‹åŒ–ï¼ˆæ¥è¿‘ GPT-2 çš„ç®€å•æ–¹æ¡ˆï¼‰
        self.apply(self._init_weights)

    def _init_weights(self, module):
        if isinstance(module, nn.Linear):
            nn.init.normal_(module.weight, mean=0.0, std=0.02)
            if module.bias is not None:
                nn.init.zeros_(module.bias)
        elif isinstance(module, nn.Embedding):
            nn.init.normal_(module.weight, mean=0.0, std=0.02)

    def forward(self, idx: torch.Tensor, targets: torch.Tensor | None = None):
        """
        idx: [B,T] çš„ token id
        targets: [B,T] çš„æ ‡ç­¾ï¼ˆå¯é€‰ï¼›ä¼ å…¥åˆ™è¿”å›äº¤å‰ç†µ lossï¼‰
        """
        B, T = idx.size()
        assert T <= self.config.block_size, "åºåˆ—é•¿åº¦è¶…è¿‡äº† block_size"

        if self.config.debug:
            print(f"\n=== Forward: idx {idx.shape} ===")

        # æ„é€ ä½ç½®ç´¢å¼• [0..T-1]
        pos = torch.arange(0, T, dtype=torch.long, device=idx.device).unsqueeze(0)  # [1,T]

        # è¯åµŒå…¥ + ä½ç½®åµŒå…¥
        tok_emb = self.wte(idx)        # [B,T,d]
        pos_emb = self.wpe(pos)        # [1,T,d]ï¼Œä¼šå¹¿æ’­åˆ° [B,T,d]
        x = self.drop(tok_emb + pos_emb)
        if self.config.debug:
            print(f"[Emb] tok_emb: {tok_emb.shape}, pos_emb: {pos_emb.shape}, x: {x.shape}")

        # å †å çš„ Transformer blocks
        for i, block in enumerate(self.h, start=1):
            if self.config.debug:
                print(f"\n-- Block {i} --")
            x = block(x)

        # æœ€ç»ˆ LN + è¯è¡¨å¤´
        x = self.ln_f(x)               # [B,T,d]
        logits = self.lm_head(x)       # [B,T,vocab]
        if self.config.debug:
            print(f"\n[Head] logits: {logits.shape}")

        # è®­ç»ƒæ—¶å¯ç›´æ¥ç®— CE loss
        loss = None
        if targets is not None:
            # æŠŠ (B,T,V) æ”¹æˆ (B*T,V) ä¸ (B*T,)
            loss = F.cross_entropy(logits.view(-1, logits.size(-1)), targets.view(-1))

        return logits, loss

    @torch.no_grad()
    def generate(self, idx: torch.Tensor, max_new_tokens: int, temperature: float = 1.0, top_k: int | None = None):
        """
        æœ´ç´ è‡ªå›å½’ç”Ÿæˆï¼ˆä¸åš KV Cacheï¼Œé€‚åˆå°è§„æ¨¡æ¼”ç¤ºï¼‰
        idx: [B,T] åˆå§‹æç¤º
        """
        for _ in range(max_new_tokens):
            # åªä¿ç•™æœ€è¿‘ block_size ä¸ª token
            idx_cond = idx[:, -self.config.block_size:]
            logits, _ = self(idx_cond)
            # å–æœ€åä¸€ä¸ªæ—¶é—´æ­¥çš„åˆ†å¸ƒ
            logits = logits[:, -1, :] / temperature
            if top_k is not None:
                v, _ = torch.topk(logits, min(top_k, logits.size(-1)))
                thresh = v[:, -1].unsqueeze(-1)
                logits = torch.where(logits < thresh, torch.full_like(logits, float("-inf")), logits)
            probs = F.softmax(logits, dim=-1)
            next_id = torch.multinomial(probs, num_samples=1)  # é‡‡æ ·ä¸€ä¸ª token
            idx = torch.cat([idx, next_id], dim=1)
        return idx

    # å°æ¨¡å‹ + å°è¯è¡¨ï¼Œæ–¹ä¾¿ CPU/GPU éƒ½èƒ½è·‘
